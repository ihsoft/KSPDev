public static void ToLocal() {
  var parent = new GameObject();
  parent.transform.position = Vector3.one;
  parent.transform.rotation = Quaternion.LookRotation(Vector3.up);
  var pr = new PosAndRot(Vector3.one, new Vector3(90, 0, 0));
